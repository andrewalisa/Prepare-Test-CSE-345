CREATE TABLE STUDENT (
STU_ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
STU_FNAME VARCHAR(100) NOT NULL,
STU_LNAME VARCHAR(100) NOT NULL,
STU_ADDRESS VARCHAR(255) NOT NULL,
STU_CITY VARCHAR(255) NOT NULL,
STU_STATE VARCHAR(2) NOT NULL,
STU_ZIP VARCHAR(5) NOT NULL,
STU_EMAIL VARCHAR(255) NOT NULL,
STU_PASSWORD VARCHAR(255) NOT NULL
);

CREATE TABLE TEMP_STUDENT (
STU_ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
STU_FNAME VARCHAR(100) NOT NULL,
STU_LNAME VARCHAR(100) NOT NULL,
STU_ADDRESS VARCHAR(255) NOT NULL,
STU_CITY VARCHAR(255) NOT NULL,
STU_ZIP VARCHAR(5) NOT NULL,
STU_EMAIL VARCHAR(255) NOT NULL,
STU_PASSWORD VARCHAR(255) NOT NULL,
STU_KEY VARCHAR(255) NOT NULL
);

CREATE TABLE TEST (
TEST_ID INT NOT NULL PRIMARY KEY,
STU_ID INT FOREIGN KEY(STU_ID) REFERENCES(STUDENT) //Look at this syntax
TEST_DATE DATE
TEST_TIME TIME 
 );
